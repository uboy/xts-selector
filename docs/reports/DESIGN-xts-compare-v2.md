# Design: xts_compare v2 — Deep Analysis & Extended Reporting

**Date**: 2026-03-28
**Status**: Ready for implementation
**Depends on**: xts_compare v1 (implemented)

---

## Обзор

xts_compare v1 парсит только `summary_report.xml` и классифицирует переходы PASS/FAIL/BLOCKED.
Реальные XTS-архивы содержат 5 дополнительных источников данных, которые v1 полностью игнорирует.

Этот документ описывает 10 задач по расширению xts_compare для:
- Глубокого анализа причин ошибок (crash vs timeout vs assertion)
- Кластеризации root cause по normalized message
- Интеграции с selector (файл→регрессия)
- HTML-отчёта для команды
- Парсинга task_info.record, data.js и crash-логов
- Performance regression detection
- Module health scoring

```python
# src/arkui_xts_selector/xts_compare/error_analysis.py

"""
Failure type classification from test messages, data.js error fields,
and optional crash/module_run logs.
"""

from __future__ import annotations

import re
from .models import FailureType

# Pattern priority: first match wins.
_CRASH_PATTERNS = [
    re.compile(r"App died", re.IGNORECASE),
    re.compile(r"SIGSEGV|SIGABRT|SIGBUS|SIGFPE|SIGILL", re.IGNORECASE),
    re.compile(r"cppcrash|jscrash|appfreeze", re.IGNORECASE),
    re.compile(r"Signal:\s*SIG", re.IGNORECASE),
    re.compile(r"Process\s+died|process\s+crash", re.IGNORECASE),
]

_TIMEOUT_PATTERNS = [
    re.compile(r"ShellCommandUnresponsiveException", re.IGNORECASE),
    re.compile(r"timeout|timed?\s*out", re.IGNORECASE),
    re.compile(r"unresponsive", re.IGNORECASE),
    re.compile(r"waited\s+\d+\s*ms", re.IGNORECASE),
]

_ASSERTION_PATTERNS = [
    re.compile(r"expected\s+.+\s+but\s+got", re.IGNORECASE),
    re.compile(r"assert(True|False|Equal|NotEqual|Null|Undefined)", re.IGNORECASE),
    re.compile(r"expect\(.+\)\.(to|not)", re.IGNORECASE),
    re.compile(r"assertion\s+(failed|error)", re.IGNORECASE),
    re.compile(r"comparison\s+failed", re.IGNORECASE),
]

_CAST_PATTERNS = [
    re.compile(r"cannot\s+be\s+cast\s+to", re.IGNORECASE),
    re.compile(r"type\s+mismatch", re.IGNORECASE),
    re.compile(r"is\s+not\s+a\s+function", re.IGNORECASE),
]

_RESOURCE_PATTERNS = [
    re.compile(r"out\s+of\s+memory|OOM", re.IGNORECASE),
    re.compile(r"no\s+space\s+left", re.IGNORECASE),
    re.compile(r"permission\s+denied", re.IGNORECASE),
    re.compile(r"ENOMEM|ENOSPC|EACCES", re.IGNORECASE),
]


def classify_failure(
    message: str,
    module_error: str = "",
) -> FailureType:
    """
    Classify a test failure based on message and optional module-level error.

    Args:
        message: Test case failure message (from XML or data.js)
        module_error: Module-level error from data.js "error" field (e.g. "App died")

    Returns:
        FailureType enum value

    The combined text (message + module_error) is matched against
    pattern groups in priority order: CRASH > TIMEOUT > CAST_ERROR > ASSERTION > RESOURCE.
    """
    combined = f"{message} {module_error}".strip()
    if not combined:
        return FailureType.UNKNOWN_FAIL

    for pattern in _CRASH_PATTERNS:
        if pattern.search(combined):
            return FailureType.CRASH

    for pattern in _TIMEOUT_PATTERNS:
        if pattern.search(combined):
            return FailureType.TIMEOUT

    for pattern in _CAST_PATTERNS:
        if pattern.search(combined):
            return FailureType.CAST_ERROR

    for pattern in _ASSERTION_PATTERNS:
        if pattern.search(combined):
            return FailureType.ASSERTION

    for pattern in _RESOURCE_PATTERNS:
        if pattern.search(combined):
            return FailureType.RESOURCE

    return FailureType.UNKNOWN_FAIL


def normalize_failure_message(raw: str) -> tuple[str, str]:
    """
    Split raw message into (short_message, detail).

    - Strips common prefixes ("Error: ", "AssertionError: ")
    - Separates stack trace from main message
    - Truncates to first meaningful line

    Returns:
        (short_message, detail_or_stack_trace)
    """
    if not raw:
        return ("", "")

    lines = raw.strip().split("\n")
    first_line = lines[0].strip()

    # Strip common prefixes
    for prefix in ("Error: ", "AssertionError: ", "TypeError: ", "ReferenceError: "):
        if first_line.startswith(prefix):
            first_line = first_line[len(prefix):]
            break

    detail = "\n".join(lines[1:]).strip() if len(lines) > 1 else ""
    return (first_line, detail)
```

### Алгоритм кластеризации

```python
# error_analysis.py

import hashlib


def _normalize_for_clustering(message: str) -> str:
    """
    Normalize message for fingerprinting:
    - Strip timestamps, PIDs, addresses
    - Lowercase
    - Collapse whitespace
    - Remove variable parts (hex addresses, line numbers, UUIDs)
    """
    import re
    text = message.lower()
    # Remove hex addresses (0x0069fffc)
    text = re.sub(r"0x[0-9a-f]+", "0xADDR", text)
    # Remove PIDs, line numbers
    text = re.sub(r"\b\d{4,}\b", "NUM", text)
    # Remove UUIDs
    text = re.sub(r"[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}", "UUID", text)
    # Collapse whitespace
    text = re.sub(r"\s+", " ", text).strip()
    return text


def _fingerprint(normalized: str) -> str:
    """SHA256 of normalized message, truncated to 16 hex chars."""
    return hashlib.sha256(normalized.encode()).hexdigest()[:16]


def cluster_failures(
    transitions: list[TestTransition],
) -> list[RootCauseCluster]:
    """
    Group failed transitions by normalized message fingerprint.

    Returns clusters sorted by count (most common first).
    """
    clusters: dict[str, RootCauseCluster] = {}

    for t in transitions:
        if t.target_outcome not in (TestOutcome.FAIL, TestOutcome.ERROR):
            continue

        msg = t.target_message or ""
        if not msg:
            msg = "(no message)"

        ft = classify_failure(msg)
        normalized = _normalize_for_clustering(msg)
        fp = _fingerprint(normalized)

        if fp not in clusters:
            clusters[fp] = RootCauseCluster(
                fingerprint=fp,
                failure_type=ft,
                canonical_message=msg,
            )

        cluster = clusters[fp]
        cluster.count += 1
        cluster.test_identities.append(t.identity)
        if t.identity.module not in cluster.modules_affected:
            cluster.modules_affected.append(t.identity.module)
        if len(cluster.example_messages) < 3 and msg not in cluster.example_messages:
            cluster.example_messages.append(msg)
        # Keep shortest non-empty as canonical
        if msg and (not cluster.canonical_message or len(msg) < len(cluster.canonical_message)):
            cluster.canonical_message = msg

    return sorted(clusters.values(), key=lambda c: -c.count)
```

```python
# parse.py

def parse_data_js(data_js_path: Path) -> dict:
    """
    Parse static/data.js.

    Format: window.reportData = { ... }
    Returns the parsed JSON dict.
    """
    import json
    text = data_js_path.read_text(encoding="utf-8", errors="replace")
    # Strip the "window.reportData = " prefix
    prefix = "window.reportData = "
    idx = text.find(prefix)
    if idx == -1:
        return {}
    json_str = text[idx + len(prefix):]
    # May have trailing semicolons or whitespace
    json_str = json_str.rstrip().rstrip(";")
    return json.loads(json_str)


def _find_data_js(directory: Path) -> Path | None:
    """Find static/data.js in archive directory."""
    for rel in ("static/data.js",):
        candidate = directory / rel
        if candidate.is_file():
            return candidate
    for root, _dirs, files in os.walk(directory):
        for name in files:
            if name == "data.js":
                return Path(root) / name
    return None


def _extract_module_infos(data: dict) -> dict[str, ModuleInfo]:
    """Extract ModuleInfo from parsed data.js."""
    result = {}
    for m in data.get("modules", []):
        name = m.get("name", "")
        if not name:
            continue
        logs = m.get("logs", {})
        result[name] = ModuleInfo(
            name=name,
            error=m.get("error", ""),
            time_s=float(m.get("time", 0)),
            tests=int(m.get("tests", 0)),
            passed=int(m.get("passed", 0)),
            failed=int(m.get("failed", 0)),
            blocked=int(m.get("blocked", 0)),
            passing_rate=m.get("passingrate", ""),
            log_refs=dict(logs) if isinstance(logs, dict) else {},
        )
    return result
```

### Парсинг

```python
# parse.py

def parse_task_info(path: Path) -> dict:
    """
    Parse task_info.record JSON.

    Returns:
        {
            "session_id": str,
            "unsuccessful": dict[str, list[tuple[str, str]]],  # module → [(suite, case)]
        }
    """
    import json
    data = json.loads(path.read_text(encoding="utf-8"))

    unsuccessful: dict[str, list[tuple[str, str]]] = {}
    for module, entries in data.get("unsuccessful_params", {}).items():
        pairs = []
        for entry in entries:
            # Format: "SuiteName#CaseName"
            if "#" in entry:
                suite, case = entry.split("#", 1)
            else:
                suite, case = "", entry
            pairs.append((suite, case))
        unsuccessful[module] = pairs

    return {
        "session_id": data.get("session_id", ""),
        "unsuccessful": unsuccessful,
    }
```

```python
    signal: str = ""                # "SIGSEGV(SEGV_MAPERR)"
    reason: str = ""                # Full reason line
    pid: int = 0
```

### Парсинг

```python
# error_analysis.py

def parse_crash_log(text: str) -> CrashInfo:
    """
    Parse a HiviewDFX cppcrash log.

    Extracts: module_name, signal, reason, PID, backtrace (top 5 frames).
    """
    info = CrashInfo()

    for line in text.split("\n"):
        line = line.strip()
        if line.startswith("Module name:"):
            info.module_name = line.split(":", 1)[1].strip()
        elif line.startswith("Pid:"):
            try:
                info.pid = int(line.split(":", 1)[1].strip())
            except ValueError:
                pass
        elif line.startswith("Process life time:"):
            info.process_life_time = line.split(":", 1)[1].strip()
        elif line.startswith("Reason:"):
            info.reason = line.split(":", 1)[1].strip()
            # Extract signal from "Signal:SIGSEGV(SEGV_MAPERR)@0x..."
            match = re.search(r"Signal:\s*(SIG\w+(?:\([^)]*\))?)", info.reason)
            if match:
                info.signal = match.group(1)
        elif line.startswith("#") and len(info.top_frames) < 5:
            # Backtrace frame: "#00 pc 0009eed0 /system/lib/..."
            # Extract function name if present
            frame = line
            paren_match = re.search(r"\(([^)]+)\)", line)
            if paren_match:
                frame = f"{line.split('(')[0].strip()} → {paren_match.group(1)}"
            info.top_frames.append(frame)

    return info
```

## Задача 6: Performance Regression Detection

**Приоритет**: P2
**Эффект**: Находит тесты которые PASS→PASS но стали в 5x медленнее (pre-regression)
**Сложность**: 2-3 часа
**Файлы**: `compare.py`, `models.py`, `format_terminal.py`, `cli.py`

### Новая модель

```python
# models.py
@dataclass
class PerformanceChange:
    """A test whose execution time changed significantly."""
    identity: TestIdentity
    base_time_ms: float
    target_time_ms: float
    delta_ms: float             # target - base
    ratio: float                # target / base (inf if base was 0)
    outcome_stable: bool        # True if outcome didn't change
```

### Алгоритм

```python
# compare.py

def detect_performance_regressions(
    base_results: dict[TestIdentity, TestResult],
    target_results: dict[TestIdentity, TestResult],
    min_delta_ms: float = 1000.0,       # minimum absolute change
    min_ratio: float = 3.0,             # minimum slowdown ratio
) -> list[PerformanceChange]:
    """
    Find tests with significant execution time changes.

    A test is flagged when:
    - Both runs have time > 0
    - abs(delta) >= min_delta_ms AND ratio >= min_ratio

    Returns sorted by ratio (biggest slowdown first).
    """
    changes = []
    common = set(base_results) & set(target_results)

    for identity in common:
        b = base_results[identity]
        t = target_results[identity]
        if b.time_ms <= 0 or t.time_ms <= 0:
            continue

        delta = t.time_ms - b.time_ms
        ratio = t.time_ms / b.time_ms

        if abs(delta) >= min_delta_ms and ratio >= min_ratio:
            changes.append(PerformanceChange(
                identity=identity,
                base_time_ms=b.time_ms,
                target_time_ms=t.time_ms,
                delta_ms=delta,
                ratio=ratio,
                outcome_stable=(b.outcome == t.outcome),
            ))

    return sorted(changes, key=lambda c: -c.ratio)
```

## Задача 7: Module Health Scoring

**Приоритет**: P2
**Эффект**: Быстрый overview — какие модули в хорошем состоянии, какие в плохом
**Сложность**: 1-2 часа
**Файлы**: `compare.py`, `models.py`, `format_terminal.py`

### Формула

```python
def compute_module_health(mc: ModuleComparison) -> float:
    """
    Compute health score 0.0-100.0 for a module.

    Factors:
    - Base: pass_rate = pass / total
    - Penalty: regressions * 10 (each regression is very bad)
    - Penalty: crashes * 5 (entire module may be down)
    - Bonus: improvements * 2 (getting better)
    """
    total = sum(mc.counts.values())
    if total == 0:
        return 100.0

    pass_count = mc.counts.get("STABLE_PASS", 0) + mc.counts.get("IMPROVEMENT", 0) + mc.counts.get("NEW_PASS", 0)
    base_score = (pass_count / total) * 100.0

    regression_penalty = mc.counts.get("REGRESSION", 0) * 10
    new_fail_penalty = mc.counts.get("NEW_FAIL", 0) * 5
    improvement_bonus = mc.counts.get("IMPROVEMENT", 0) * 2

    score = base_score - regression_penalty - new_fail_penalty + improvement_bonus
    return max(0.0, min(100.0, score))
```
